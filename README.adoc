== Docker
[source,bash]
----
# Build container
docker build -t asciidoctor-lists .
# Run Docker
docker run -it --rm -v $(pwd):/work asciidoctor-lists
# Build sample
asciidoctor-pdf -r asciidoctor-lists samples/list-of-figures.adoc
----

----

gem install asciidoctor-lists-0.0.2.gem
